
git clone   -- клонирует репозиторий
git add .   -- говорит гиту отслеживать все изменения в вашем локальном репозитории
git commit -m 'message'   -- "закомитеть"
git push   -- "запушить"
git status   -- показать статус локального репозитория
git pull   -- стянуть все обновления с локального репозитория
git log   -- показать историю коммитов
git branch   -- показать список всех локальных веток
git checkout -b 'name'   -- создать новую ветку с 'именем' и сразу в него перейти 
git checkout 'name'   -- перейти ветку с 'именем'
